La aplicación debe implementar la siguiente lógica:
	a) Crear un nuevo autor X
	b) Crear un nuevo libro X 
	c) Asociar un autor con un libro X
	d) Crear un autor y su libro en cascada X
	e) Listar libros de un autor X
	f) Listar autores de un libro X
	g) Generar un reporte personalizado de todos los autores y sus libros:
	    nombreAutor | títuloLibro



CREATE DATABASE IF NOT EXISTS biblioteca_city CHARSET utf8mb4 COLLATE utf8mb4_0900_ai_ci;

USE biblioteca_city;

CREATE TABLE auIdtor (
    id_autor SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    sex enum('H','M') DEFAULT 'M' );

CREATE TABLE libro (
    id_libro  SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(50) NOT NULL,
    fechaPublicacion DATE);

CREATE TABLE autor_libro (
    autor_id SMALLINT UNSIGNED NOT NULL,
    libro_id SMALLINT UNSIGNED NOT NULL,

    PRIMARY KEY (autor_id, libro_id),

    CONSTRAINT fk_autor_libro_autor
        FOREIGN KEY (autor_id)
        REFERENCES autor(id_autor),

    CONSTRAINT fk_autor_libro_libro
        FOREIGN KEY (libro_id)
        REFERENCES libro(id_libro)
);
